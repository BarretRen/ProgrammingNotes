
# 堆的操作
* 插入新节点：向上渗透
* 删除根节点：向下渗透


## 插入新节点
算法：

- 将新节点放在堆最后
- 向上渗透：比较新加节点与父节点的大小，进行交换，直到符合大顶堆或小顶堆的规则


## 删除根节点
算法：

- 删除根节点
- 将最后一个节点放在根节点位置
- 向下渗透：比较此节点与左右子节点的大小，进行交换，直到符合大顶堆或小顶堆的规则


# 大顶堆code

```cpp
#ifndef MAXHEAP_H
#define MAXHEAP_H

//大顶堆类
template<typename T>
class MaxHeap
{
public:
    MaxHeap(int size = 10); //默认大小为10
    virtual ~MaxHeap();

    bool isEmpty();
    void push(const T& value);
    void pop();//删除根节点
    const T& Top();//取出根节点
private:
    T* heapArray;//用数组保存堆
    int maxSize; //堆大小
    int currentSize;//当前堆里的数据个数

    //向上渗透函数
    void trickleUp(int index);
    //向下渗透函数
    void trickleDown(int index);
};
template<typename T>
MaxHeap<T>::MaxHeap(int size)
{
    if (size < 1)
        throw "heap size must be >=1";

    maxSize = size;
    currentSize = 0;
    heapArray = new T[maxSize];
}

template<typename T>
MaxHeap<T>::~MaxHeap()
{
    delete[] heapArray;
}

template<typename T>
bool MaxHeap<T>::isEmpty()
{
    return currentSize == 0;
}

template<typename T>
void MaxHeap<T>::push(const T& value)
{
    if (currentSize == maxSize)
        throw "heap is full";

    heapArray[currentSize] = value; //先将新节点放在最后
    trickleUp(currentSize);//向上渗透
    currentSize++;
}

template<typename T>
void MaxHeap<T>::pop()
{
    if (currentSize == 0)
        return;

    heapArray[0] = heapArray[currentSize - 1];
    currentSize--;
    trickleDown(0);
}

template<typename T>
const T& MaxHeap<T>::Top()
{
    return heapArray[0]; //返回第一个的值
}

template<typename T>
void MaxHeap<T>::trickleUp(int index)
{
    //当前节点的父节点index是：(index - 1)/2
    int parent = (index - 1) / 2;
    T tmp = heapArray[index];

    //循环条件，index最小到根节点，父节点小于当前值时停止
    while(index > 0 && heapArray[parent] < tmp)
    {
        heapArray[index] = heapArray[parent]; //小的移到后面，交换位置
        index = parent;
        parent = (parent - 1) / 2;
    }

    heapArray[index] = tmp;
}

template<typename T>
void MaxHeap<T>::trickleDown(int index)
{
    T tmp = heapArray[index];
    int leftChild;
    int rightChild;
    int largerChild;

    while (index < currentSize / 2)
    {
        leftChild = 2 * index + 1;
        rightChild = leftChild + 1;
        //因为父节点要比两个子节点都大，所以先找到最大的那个子节点
        if (rightChild < currentSize && heapArray[leftChild] < heapArray[rightChild])
            largerChild = rightChild;
        else
            largerChild = leftChild;

        //比较父节点与较大子节点的大小
        if (tmp >= heapArray[largerChild])
            break;
        else
            heapArray[index] = heapArray[largerChild];

        index = largerChild;
    }

    heapArray[index] = tmp;
}
#endif // MAXHEAP_H
```
